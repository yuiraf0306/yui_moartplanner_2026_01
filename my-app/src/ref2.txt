const getWeekResetReference = (dateObj) => {
  const d = new Date(dateObj);
  d.setHours(d.getHours() - 23);
  const dayOfWeek = d.getDay();
  const diff = d.getDate() - dayOfWeek;
  d.setDate(diff);
  d.setHours(0, 0, 0, 0);
  return d; 
};

const calculateDateForWeekday = (weekdayId, referenceSunday) => {
    let targetDay = parseInt(weekdayId, 10);
    if (targetDay === 0) targetDay = 7; 
    const d = new Date(referenceSunday);
    d.setDate(d.getDate() + targetDay);
    const year = d.getFullYear();
    const month = String(d.getMonth() + 1).padStart(2, '0');
    const day = String(d.getDate()).padStart(2, '0');
    return `${year}-${month}-${day}`;
};

// 圖片壓縮輔助函數
const compressImage = (file, maxSize = 300) => new Promise((resolve) => { 
  const reader = new FileReader(); reader.readAsDataURL(file); 
  reader.onload = (e) => { 
    const img = new Image(); img.src = e.target.result; 
    img.onload = () => { 
      const canvas = document.createElement('canvas'); 
      const ctx = canvas.getContext('2d'); 
      if (maxSize === 300) { 
        canvas.width = maxSize; canvas.height = maxSize; 
        const m = Math.min(img.width, img.height); 
        ctx.drawImage(img, (img.width-m)/2, (img.height-m)/2, m, m, 0, 0, maxSize, maxSize); 
      } else { 
        let w = img.width; let h = img.height; 
        if (w > h && w > maxSize) { h *= maxSize / w; w = maxSize; } 
        else if (h > w && h > maxSize) { w *= maxSize / h; h = maxSize; } 
        canvas.width = w; canvas.height = h; ctx.drawImage(img, 0, 0, w, h); 
      } 
      resolve(canvas.toDataURL('image/jpeg', 0.8)); 
    } 
  } 
});

// 防抖函式，用於減少儲存頻率
const debounce = (func, wait) => {
  let timeout;
  return function executedFunction(...args) {
    const later = () => {
      clearTimeout(timeout);
      func(...args);
    };
    clearTimeout(timeout);
    timeout = setTimeout(later, wait);
  };
};

// ==========================================
const ScrapbookPhoto = ({ photo, index, isLocked, onUpdatePhoto, onPreview, containerRef }) => {
  const [localTransform, setLocalTransform] = useState({
    x: photo.x || (index * 20 + 20), 
    y: photo.y || (index * 20 + 20), 
    scale: photo.scale || 1, 
    panX: photo.panX || 0, 
    panY: photo.panY || 0, 
    cropScale: photo.cropScale || 1
  });
  const [isDragging, setIsDragging] = useState(false);
  const [isCropping, setIsCropping] = useState(false);
  const [isCropDragging, setIsCropDragging] = useState(false);
  
  const dragRef = useRef(null);
  const startPosRef = useRef({ x: 0, y: 0, initialX: 0, initialY: 0 });
  const pinchRef = useRef({ initialDist: 0, initialScale: 1, isPinching: false });
  const cropStartPosRef = useRef({ x: 0, y: 0, initialPanX: 0, initialPanY: 0 });

  useEffect(() => {
    if (!isDragging && !isCropping && !isCropDragging) {
      setLocalTransform({ 
        x: photo.x || (index * 20 + 20), 
        y: photo.y || (index * 20 + 20), 
        scale: photo.scale || 1, 
        panX: photo.panX || 0, 
        panY: photo.panY || 0, 
        cropScale: photo.cropScale || 1 
      });
    }
  }, [photo, index, isDragging, isCropping, isCropDragging]);

  const handlePointerDown = (e) => {
    if (isLocked || isCropping || (e.pointerType === 'mouse' && e.button !== 0)) return;
    e.stopPropagation(); 
    e.target.setPointerCapture(e.pointerId);
    setIsDragging(true);
    startPosRef.current = { x: e.clientX, y: e.clientY, initialX: localTransform.x, initialY: localTransform.y };
  };

  const handlePointerMove = (e) => {
    if (!isDragging || isLocked || isCropping) return;
    e.stopPropagation();
    const dx = e.clientX - startPosRef.current.x; 
    const dy = e.clientY - startPosRef.current.y;
    setLocalTransform(prev => ({ ...prev, x: startPosRef.current.initialX + dx, y: startPosRef.current.initialY + dy }));
  };

  const handlePointerUp = (e) => {
    if (!isDragging) {
      if (isLocked && Math.abs(e.clientX - startPosRef.current.x) < 5 && Math.abs(e.clientY - startPosRef.current.y) < 5) {
         onPreview(photo.url);
      }
    }
    if (isDragging && !isLocked && !isCropping) {
      e.target.releasePointerCapture(e.pointerId); 
      onUpdatePhoto(index, { x: localTransform.x, y: localTransform.y, scale: localTransform.scale });
      
      setTimeout(() => {
         setIsDragging(false);
      }, 50);
    }
  };

  const handleTouchStart = (e) => {
    if (isLocked || isCropping) return;
    if (e.touches.length === 2) {
      e.stopPropagation(); e.preventDefault(); setIsDragging(false); pinchRef.current.isPinching = true;
      pinchRef.current.initialDist = Math.hypot(e.touches[0].clientX - e.touches[1].clientX, e.touches[0].clientY - e.touches[1].clientY);
      pinchRef.current.initialScale = localTransform.scale;
    }
  };

  const handleTouchMove = (e) => {
    if (!pinchRef.current.isPinching || isLocked || isCropping) return;
    if (e.touches.length === 2) {
      e.stopPropagation(); e.preventDefault();
      const dist = Math.hypot(e.touches[0].clientX - e.touches[1].clientX, e.touches[0].clientY - e.touches[1].clientY);
      const newScale = Math.max(0.3, Math.min(pinchRef.current.initialScale * (dist / pinchRef.current.initialDist), 3));
      setLocalTransform(prev => ({ ...prev, scale: newScale }));
    }
  };

  const handleTouchEnd = () => {
    if (pinchRef.current.isPinching) {
      pinchRef.current.isPinching = false;
      onUpdatePhoto(index, { scale: localTransform.scale });
    }
  };

  const handleCropPointerDown = (e) => {
    e.stopPropagation(); 
    e.target.setPointerCapture(e.pointerId);
    setIsCropDragging(true);
    cropStartPosRef.current = { x: e.clientX, y: e.clientY, initialPanX: localTransform.panX, initialPanY: localTransform.panY };
  };

  const handleCropPointerMove = (e) => {
    if (!isCropDragging) return; 
    e.stopPropagation();
    setLocalTransform(prev => ({ 
        ...prev, 
        panX: cropStartPosRef.current.initialPanX + (e.clientX - cropStartPosRef.current.x) / prev.cropScale, 
        panY: cropStartPosRef.current.initialPanY + (e.clientY - cropStartPosRef.current.y) / prev.cropScale 
    }));
  };

  const handleCropPointerUp = (e) => {
    if (isCropDragging) {
        setIsCropDragging(false);
        e.target.releasePointerCapture(e.pointerId);
    }
  };
  
  const handleCropTouchStart = (e) => {
     if (e.touches.length === 2) {
      e.stopPropagation(); e.preventDefault(); pinchRef.current.isPinching = true;
      pinchRef.current.initialDist = Math.hypot(e.touches[0].clientX - e.touches[1].clientX, e.touches[0].clientY - e.touches[1].clientY);
      pinchRef.current.initialScale = localTransform.cropScale;
    }
  };
  
  const handleCropTouchMove = (e) => {
     if (!pinchRef.current.isPinching) return;
     if (e.touches.length === 2) {
      e.stopPropagation(); e.preventDefault();
      const dist = Math.hypot(e.touches[0].clientX - e.touches[1].clientX, e.touches[0].clientY - e.touches[1].clientY);
      setLocalTransform(prev => ({ ...prev, cropScale: Math.max(0.5, Math.min(pinchRef.current.initialScale * (dist / pinchRef.current.initialDist), 5)) }));
    }
  };

  if (isCropping) {
    return createPortal(
      <div className="absolute inset-0 z-[200] bg-black/90 flex flex-col items-center justify-center touch-none" 
           onPointerDown={handleCropPointerDown} 
           onPointerMove={handleCropPointerMove} 
           onPointerUp={handleCropPointerUp}
           onPointerCancel={handleCropPointerUp}
           onTouchStart={handleCropTouchStart} 
           onTouchMove={handleCropTouchMove} 
           onTouchEnd={() => { pinchRef.current.isPinching = false; }}
           onClick={(e) => { e.stopPropagation(); setIsCropping(false); setLocalTransform(p => ({ ...p, panX: photo.panX||0, panY: photo.panY||0, cropScale: photo.cropScale||1 })); }}
      >
        <p className="text-white/50 text-xs font-bold tracking-widest uppercase mb-6 z-20 pointer-events-none">Drag to Pan • Pinch to Zoom</p>

        <div className="relative w-64 h-64 flex items-center justify-center cursor-move" onClick={e => e.stopPropagation()}>
           <img 
             src={photo.url} 
             className="absolute pointer-events-none touch-none opacity-40" 
             style={{ 
               top: '50%', left: '50%', minWidth: '100%', minHeight: '100%', width: 'auto', height: 'auto', maxWidth: 'none', maxHeight: 'none', 
               transform: `translate(calc(-50% + ${localTransform.panX}px), calc(-50% + ${localTransform.panY}px)) scale(${localTransform.cropScale})` 
             }} 
             alt="background" 
           />

           <div className="absolute inset-0 overflow-hidden pointer-events-none">
             <img 
                 src={photo.url} 
                 className="absolute pointer-events-none touch-none" 
                 style={{ 
                   top: '50%', left: '50%', minWidth: '100%', minHeight: '100%', width: 'auto', height: 'auto', maxWidth: 'none', maxHeight: 'none', 
                   transform: `translate(calc(-50% + ${localTransform.panX}px), calc(-50% + ${localTransform.panY}px)) scale(${localTransform.cropScale})` 
                 }} 
                 alt="cropping" 
               />
           </div>

           <div className="absolute inset-0 border-2 border-white/80 shadow-[0_0_0_9999px_rgba(0,0,0,0.6)] pointer-events-none rounded">
               <div className="absolute inset-0 grid grid-cols-3 grid-rows-3 pointer-events-none opacity-30 z-10">
                   <div className="border-r border-b border-white"></div><div className="border-r border-b border-white"></div><div className="border-b border-white"></div>
                   <div className="border-r border-b border-white"></div><div className="border-r border-b border-white"></div><div className="border-b border-white"></div>
                   <div className="border-r border-white"></div><div className="border-r border-white"></div><div></div>
               </div>
           </div>
        </div>
        
        <div className="flex gap-4 mt-8 z-20 pointer-events-auto" onClick={e => e.stopPropagation()}>
           <button onClick={(e) => { e.stopPropagation(); setIsCropping(false); setLocalTransform(p => ({ ...p, panX: photo.panX||0, panY: photo.panY||0, cropScale: photo.cropScale||1 })); }} className="w-12 h-12 rounded-full bg-slate-800 text-white flex items-center justify-center shadow-lg active:scale-95 transition-transform"><X size={20}/></button>
           <button onClick={(e) => { e.stopPropagation(); onUpdatePhoto(index, { panX: localTransform.panX, panY: localTransform.panY, cropScale: localTransform.cropScale }); setTimeout(() => setIsCropping(false), 50); }} className="w-12 h-12 rounded-full bg-purple-600 text-white flex items-center justify-center shadow-[0_0_15px_rgba(168,85,247,0.5)] active:scale-95 transition-transform"><Check size={20}/></button>
        </div>
      </div>,
      containerRef.current || document.body
    );
  }

  return (
    <div 
      ref={dragRef} 
      className={`absolute transition-shadow duration-200 ${isDragging ? 'z-30 shadow-2xl scale-105 opacity-90' : 'z-10 shadow-md'} ${!isLocked ? 'cursor-grab active:cursor-grabbing' : 'cursor-pointer'} touch-none`} 
      style={{ transform: `translate(${localTransform.x}px, ${localTransform.y}px) scale(${localTransform.scale})`, width: '120px', height: '120px', transformOrigin: 'top left' }} 
      onPointerDown={handlePointerDown} 
      onPointerMove={handlePointerMove} 
      onPointerUp={handlePointerUp} 
      onPointerCancel={handlePointerUp} 
      onTouchStart={handleTouchStart} 
      onTouchMove={handleTouchMove} 
      onTouchEnd={handleTouchEnd} 
      onTouchCancel={handleTouchEnd}
    >
      <div className={`w-full h-full overflow-hidden bg-white p-1 rounded-sm ${!isLocked ? 'ring-2 ring-orange-400 border border-dashed border-orange-200' : ''}`}>
        <div className="w-full h-full overflow-hidden relative bg-slate-100 flex items-center justify-center pointer-events-none">
           <div style={{ width: 256, height: 256, position: 'absolute', top: '50%', left: '50%', transform: 'translate(-50%, -50%) scale(0.4375)' }}>
               <img 
                 src={photo.url} 
                 className="absolute pointer-events-none touch-none" 
                 style={{ 
                   top: '50%', left: '50%', minWidth: '100%', minHeight: '100%', width: 'auto', height: 'auto', maxWidth: 'none', maxHeight: 'none', 
                   transform: `translate(calc(-50% + ${localTransform.panX}px), calc(-50% + ${localTransform.panY}px)) scale(${localTransform.cropScale})` 
                 }} 
                 alt="scrapbook item" 
               />
           </div>
        </div>
      </div>
      {!isLocked && (
          <div 
            className="absolute -bottom-2 -right-2 w-8 h-8 bg-orange-500 rounded-full flex items-center justify-center text-white shadow-lg shadow-orange-900/50 pointer-events-auto cursor-pointer hover:scale-110 active:scale-95 transition-transform z-20"
            onPointerDown={(e) => {
              e.stopPropagation();
              setIsCropping(true);
            }}
          >
            <Crop size={14}/>
          </div>
      )}
    </div>
  );
};

  const days = []; const year = currentDate.getFullYear(); const month = currentDate.getMonth(); const first = new Date(year, month, 1); const last = new Date(year, month+1, 0); for(let i=0; i<first.getDay(); i++) days.push(null); for(let i=1; i<=last.getDate(); i++) days.push(new Date(year, month, i));

            {view!=='month'?<button onClick={handleBack} className="flex items-center gap-1 text-purple-200 bg-white/5 hover:bg-white/10 px-3 py-1.5 rounded-full border border-white/10"><ArrowLeft size={16}/><span className="text-xs font-bold">BACK</span></button>:<div className="flex items-center gap-2 text-purple-200"><Palette size={20} className="drop-shadow-[0_0_8px_rgba(168,85,247,0.8)]"/><span className="text-xs font-black tracking-[0.2em] uppercase">Mo Art Planner</span></div>}

            ) : (
